define limit = 100000
define count = 0
define i = 0

repeat
	define j = 2
	at innerloop
		if < j i then else do prime end
		define r = % i j
		if = r 0 then do composite end del r
		set j = + j 1 del j
		do innerloop del innerloop
at prime 
	set count = + count 1
at composite
	set i = + i 1
	while < i limit del i

eoi
