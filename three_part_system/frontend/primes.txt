( a prime number program to count 
  the number of primes less than a 
  given limit, in this case 1000.
  written 1202512092.054922 by dwrr
)

define limit = 1000
define count = 0
define i = 0

repeat
	define j = 2
	at innerloop
		if < j i else do prime end
		define r = % i j
		if = r 0 do composite else end del r
		set j = + j 1 del j
		do innerloop del innerloop
at prime 
	set count = + count 1
at composite
	set i = + i 1
	while < i limit del i

eoi
