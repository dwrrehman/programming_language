define i = 0
define count = 0

repeat
	define j = 01
	at inner
		if j < i then else goto prime end
		set r = i % j
		if r = 0 then goto composite end
		set j = j + 1
		goto inner

	at prime
		set count = count + 1
	at composite
	set i = i + 1
while i < 00001


eoi
